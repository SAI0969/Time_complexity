Insertion Sort

Algorithm Code Time Complexity
1. Start

2. Read n (number of elements)

3. Read array elements

4. Assume first element is sorted

5. Run a loop i = 1 to n - 1
key = arr[i]

j = i - 1

    While j >= 0 and arr[j] > key :
        arr[j+ 1] = arr[j]
        j = j - 1

    arr[j + 1] = key

6. Stop
 
Without Function:    
 
#include <stdio.h>

int main()
{
    int arr[100], n, i, j, key;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter array elements:\n");
    for(i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    for(i = 1; i < n; i++)
    {
        key = arr[i];
        j = i - 1;

        while(j >= 0 && arr[j] > key)
        {
            arr[j + 1] = arr[j];
            j--;
        }

        arr[j + 1] = key;
    }

    printf("Sorted array:\n");
    for(i = 0; i < n; i++)
        printf("%d ", arr[i]);

    return 0;
}

With Function :
#include <stdio.h>

void insertionSort(int arr[], int n)
{
    int i, j, key;

    for(i = 1; i < n; i++)
    {
        key = arr[i];
        j = i - 1;

        while(j >= 0 && arr[j] > key)
        {
            arr[j + 1] = arr[j];
            j--;
        }

        arr[j + 1] = key;
    }
}

int main()
{
    int arr[] = {50, 20, 40, 10, 30};
    int n = 5;
    int i;

    insertionSort(arr, n);

    printf("Sorted array:\n");
    for(i = 0; i < n; i++)
        printf("%d ", arr[i]);

    return 0;
}
O(n²)

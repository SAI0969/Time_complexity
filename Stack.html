Push

Algorithm

1. Check if stack is full
2. If full, print Overflow
3. Increment top
4. Insert element at top

Code

void push(int data)
{
    if(top == MAX - 1)
    {
        printf("Stack Overflow");
        return;
    }

    top++;
    stack[top] = data;
}
 O(1)
Pop

Algorithm

1. Check if stack is empty
2. If empty, print Underflow
3. Store top Element
4. Decrement top
5. Return deleted element

Code

int pop()
{
    if(top == -1)
    {
        printf("Stack Underflow");
        return -1;
    }

    int item = stack[top];
    top--;

    return item;
}
 O(1)
Peek

Algorithm

1. Check whether stack is empty
2. Return top element

Code

int peek()
{
    if(top == -1)
    {
        printf("Stack Empty");
        return -1;
    }

    return stack[top];
}
 O(1)
Display

Algorithm

1. Check if stack is empty
2. Traverse from top to bottom
3. Print all elements

Code

void display()
{
    if(top == -1)
    {
        printf("Stack Empty");
        return;
    }

    for(int i = top; i >= 0; i--)
        printf("%d ", stack[i]);
}
 O(n)
isEmpty()

Algorithm

1. Check whether top = -1
2. If yes, stack is empty

Code

int isEmpty()
{
    if(top == -1)
        return 1;

    return 0;
}
O(1)
isFull()

Algorithm

1. Check whether top = Max - 1
2. If yes, stack is full

Code

int isFull()
{
    if(top == MAX - 1)
        return 1;

    return 0;
}
O(1)
String Reversal 
#include < stdio.h>
#include < string.h>

#define MAX 100

char stack[MAX];
int top = -1;

void push(char ch)
{
    if(top == MAX - 1)
        return;

    stack[++top] = ch;
}

char pop()
{
    if(top == -1)
        return '\0';

    return stack[top--];
}

int main()
{
    char str[100];

    printf("Enter a string: ");
    scanf("%s", str);

    int len = strlen(str);

    for(int i = 0; i < len; i++)
        push(str[i]);

    for(int i = 0; i < len; i++)
        str[i] = pop();

    printf("Reversed String = %s", str);

    return 0;
}
O(n)

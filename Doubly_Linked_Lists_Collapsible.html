Traversal (Forward)

Algorithm

1. Start from head
2. Print data
3. Move to next pointer
4. Repeat until NULL
5. Stop

Code


    void displayForward()
{
    struct Node *temp = head;

    while(temp != NULL)
    {
        printf("%d ", temp->data);
        temp = temp->next;
    }
}

Traversal (Backward)

Algorithm

1. Traverse to last node
2. Print node data
3. Move to prev pointer
4. Repeat until NULL
5. Stop

Code


void displayBackward()
{
    struct Node *temp = head;

    while(temp->next != NULL)
        temp = temp->next;

    while(temp != NULL)
    {
        printf("%d ", temp->data);
        temp = temp->prev;
    }
}
O(n)
Insertion (Beginning)

Algorithm

1. Create a new node
2. Insert data into the node
3. Set newNode->prev = NULL
4. Set newNode->next = Head
5. If list not Empty, Set head->prev = newNode
6. Update Head

Code

void insertBeginning(int data)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;
    newNode->prev = NULL;
    newNode->next = head;

    if(head != NULL)
        head->prev = newNode;

    head = newNode;
}
 O(1)
Insertion (End)

Algorithm

1. Create a new node
2. Set next = NULL
3. If list is empty, make it head
4. Traverse to last node
5. Link last node and new node

Code

void insertEnd(int data)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));
    newNode->data = data;
    newNode->next = NULL;
    if(head == NULL)
    {
        newNode->prev = NULL;
        head = newNode;
        return;
    }
    struct Node *temp = head;
    while(temp->next != NULL)
        temp = temp->next;
    temp->next = newNode;
    newNode->prev = temp;}

Insertion (At Specific Position)

Algorithm

1. Create a new node
2. Traverse to position-1
3. Connect new node between nodes
4. Update prev and next links

Code

void insertPosition(int data, int pos)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));
    newNode->data = data;
    struct Node *temp = head;
    for(int i=1;i < pos-1; i++)
        temp = temp->next;
    newNode->next = temp->next;
    newNode->prev = temp;
    if(temp->next != NULL)
        temp->next->prev = newNode;
    temp->next = newNode;
}
 O(n)
Deletion (At Beginning)

Algorithm

1. Store first node
2. Move head to next node
3. Set new head's prev as NULL
4. Delete old first node

Code


void deleteBeginning()
{
    if(head == NULL)
        return;

    struct Node *temp = head;

    head = head->next;

    if(head != NULL)
        head->prev = NULL;

    free(temp);
}
O(1)
Deletion (At End)

Algorithm

1. Traverse to last node
2. Move to previous node
3. Set prev node's next as NULL
4. Delete last node

Code


void deleteEnd()
{
    if(head == NULL)
        return;
    struct Node *temp = head;
    while(temp->next != NULL)
        temp = temp->next;
    if(temp == head)
    {
        free(head);
        head = NULL;
        return;
    }
    temp->prev->next = NULL;
    free(temp);
}
O(n)
Deletion (At Specific Position)

Algorithm

1. Traverse to desired node
2. Connect previous nodes and next nodes
3. Delete current node

Code


void deletePosition(int pos)
{
    struct Node *temp = head;

    for(int i=1;i < pos; i++)
        temp = temp->next;

    temp->prev->next = temp->next;

    if(temp->next != NULL)
        temp->next->prev = temp->prev;

    free(temp);
}
O(n)
Searching a node

Algorithm

1. Start from head
2. Compare each node with key
3. If found, return position
4. Otherwise continue
5. If end reached, element not found

Code


void search(int key)
{
    struct Node *temp = head;
    int pos = 1;
    while(temp != NULL)
    {
        if(temp->data == key)
        {
            printf("Found at position %d", pos);
            return;
        }
        temp = temp->next;
        pos++;
    }
    printf("Not Found");
}
O(n)
Updating a node

Algorithm

1. Search required node
2. Replace old value with new value

Code


void update(int oldValue, int newValue)
{
    struct Node *temp = head;

    while(temp != NULL)
    {
        if(temp->data == oldValue)
        {
            temp->data = newValue;
            return;
        }

        temp = temp->next;
    }
}
O(n)
Reversing

Algorithm

1. Traverse each node
2. Swap prev and next pointers
3. Move using swapped prev
4. Make last processed node the new head

Code


void reverse()
{
    struct Node *temp = NULL;
    struct Node *current = head;
    while(current != NULL)
    {
        temp = current->prev;
        current->prev = current->next;
        current->next = temp;
        current = current->prev;
    }
    if(temp != NULL)
        head = temp->prev;
}
O(n)
Count number of nodes

Algorithm

1. Initialize count = 0
2. Traverse entire list
3. Increment count for each node
4. Print count

Code


int countNodes()
{
    int count = 0;

    struct Node *temp = head;

    while(temp != NULL)
    {
        count++;
        temp = temp->next;
    }

    return count;
}
O(n)

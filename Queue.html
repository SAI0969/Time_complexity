Enqueue

Algorithm

1. Check if queue is full
2. If full, print Overflow
3. If queue is empty :
set front = rear = 0
4. Else:
Increment rear
5. Insert elements at rear

Code

void enqueue(int data)
{
    if(rear == MAX - 1)
    {
        printf("Queue Overflow");
        return;
    }

    if(front == -1)
        front = rear = 0;
    else
        rear++;

    queue[rear] = data;
}
 O(1)
Dequeue

Algorithm

1. Check if queue is empty
2. If empty, print Underflow
3. Store front element
4. If only one element :
Set front = rear = -1
5. Else:
Increment front
6. Return deleted element

Code

int dequeue()
{
    if(front == -1)
    {
        printf("Queue Underflow");
        return -1;
    }
    int item = queue[front];
    if(front == rear)
        front = rear = -1;
    else
        front++;
    return item;
}
 O(1)
Peek

Algorithm

1. Check whether queue is empty
2. Return front element

Code

int peek()
{
    if(front == -1)
    {
        printf("Queue Empty");
        return -1;
    }

    return queue[front];
}
 O(1)
Display

Algorithm

1. Check if queue is empty
2. Traverse from front to rear
3. Print all elements

Code

void display()
{
    if(front == -1)
    {
        printf("Queue Empty");
        return;
    }

    for(int i = front; i <= rear; i++)
        printf("%d ", queue[i]);
}
 O(n)

Size

Algorithm

1. Check if queue is empty
2. If empty return 0
3. Otherwise calculate :
Size = rear - front + 1
4. Return size

Code

int size()
{
    if(front == -1)
        return 0;

    return rear - front + 1;
}
O(1)

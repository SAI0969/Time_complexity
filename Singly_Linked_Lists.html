void display() {
    struct Node *temp = head;

    while(temp != NULL) {
        printf("%d -> ", temp->data);
        temp = temp->next;

void insertBeginning(int data)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;
    newNode->next = head;
    head = newNode;
}

Insertion (End)
Algorithm
1. Create a new node
2. Set newNode->next = NULL
3. If list is empty, make it head
4. Traverse to last node
5. Link last node to new node
Code 
void insertEnd(int data)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;
    newNode->next = NULL;

    if(head == NULL)
    {
        head = newNode;
        return;
    }

    struct Node *temp = head;

    while(temp->next != NULL)
        temp = temp->next;

    temp->next = newNode;
}

Insertion (At Specific Position)
Algorithm
1. Create a new node
2. Traverse to position-1
3. Set newNode->next = temp->next
4. Set temp->next = newNode
Code 
void insertPosition(int data, int pos)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;

    struct Node *temp = head;

    for(int i=1; i < pos-1 ;i++)
        temp = temp->next;

    newNode->next = temp->next;
    temp->next = newNode;
}

Deletion (At End)
Algorithm
1. Traverse until second-last node
2. Store last node
3. Set second-last node's next as NULL
4. Delete last node
Code 

  void deleteEnd()
{
    if(head == NULL)
        return;

    if(head->next == NULL)
    {
        free(head);
        head = NULL;
        return;
    }

    struct Node *temp = head;

    while(temp->next->next != NULL)
        temp = temp->next;

    free(temp->next);
    temp->next = NULL;
}

{

O(n)
Searching a node
Algorithm
1. Start from head
2. Compare each node with key
3. If found, return position
4. Otherwise continue
5. If end reached, element not found
Code 

void search(int key)
{
    struct Node *temp = head;
    int pos = 1;

    while(temp != NULL)
    {
        if(temp->data == key)
        {
            printf("Found at position %d", pos);
            return;
        }

        temp = temp->next;
        pos++;
    }

    printf("Not Found");
}
O(n)
Updating a node
Algorithm
1. Search required node
2. Replace old value with new value
Code 

void update(int oldValue, int newValue)
{
    struct Node *temp = head;

    while(temp != NULL)
    {
        if(temp->data == oldValue)
        {
            temp->data = newValue;
            return;
        }

        temp = temp->next;
    }
}
O(n)
Reversing
Algorithm
1. Initialize:
prev = NULL
current = head
2. Store next node
3. Reverse Link
4. Move pointers forward
5. Repeat untill end
6. Set head = prev
Code 

void reverse()
{
    struct Node *prev = NULL;
    struct Node *current = head;
    struct Node *next = NULL;

    while(current != NULL)
    {
        next = current->next;
        current->next = prev;

        prev = current;
        current = next;
    }

    head = prev;
}
O(n)
Count number of nodes
Algorithm
1. Initialize count = 0
2. Traverse entire list
3. Increment count for each node
4. Print count
Code 

int countNodes()
{
    int count = 0;

    struct Node *temp = head;

    while(temp != NULL)
    {
        count++;
        temp = temp->next;
    }

    return count;
}

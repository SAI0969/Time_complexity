void insertBeginning(int data)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;
    newNode->next = head;
    head = newNode;
}
1. Create a new node
2. Insert data into the node
3. Set newNode->next = head
4. Update head = newNode

void insertEnd(int data)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;
    newNode->next = NULL;

    if(head == NULL)
    {
        head = newNode;
        return;
    }

    struct Node *temp = head;

    while(temp->next != NULL)
        temp = temp->next;

    temp->next = newNode;
}
1. Create a new node
2. Set newNode->next = NULL
3. If list is empty, make it head
4. Traverse to last node
5. Link last node to new node
O(n)

void insertPosition(int data, int pos)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;

    struct Node *temp = head;

    for(int i=1; i < pos-1 ;i++)
        temp = temp->next;

    newNode->next = temp->next;
    temp->next = newNode;
}
1. Create a new node
2. Traverse to position-1
3. Set newNode->next = temp->next
4. Set temp->next = newNode
O(n)
